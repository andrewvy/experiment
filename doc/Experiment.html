    experiment v0.0.1 Experiment behaviour

    This module injects the lab config for working with labs. This also has the methods for creating and running lab experiments.

    Example

    defmodule Controller do
  use Experiment

  def main do
    lab("Test my new experimental function")
    |> experiment(&experimental_function/0)
    |> control(&control_function/0)
    |> perform_experiment
  end

  def control_function do
    {:ok, :foo}
  end

  def experimental_function do
    {:ok, :bar}
  end
end
