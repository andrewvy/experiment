    experiment v0.0.1 Experiment

    This module injects the lab config for working with labs. This also has the methods for creating and running lab experiments.

    Example

    defmodule Controller do

  def main do
    Experiment.new("Test my new experimental function")
    |> Experiment.experiment(&experimental_function/0)
    |> Experiment.control(&control_function/0)
    |> Experiment.perform_experiment
  end

  def control_function do
    {:ok, :foo}
  end

  def experimental_function do
    {:ok, :bar}
  end
end
